CREATE TRIGGER check_precedences
AFTER INSERT on EXAMINATION
FOR EACH ROW
WHEN (SELECT PrecedingCourse
	  FROM PRECEDENCERULE
	  WHERE FollowingCourse = new.Course) 
	  NOT IN (SELECT Course
			  FROM EXAMINATION
			  WHERE StudentID = new.StudentID AND Date < new.Date)
rollback;

CREATE TRIGGER recursive_trigger
AFTER UPDATE of MinDistanceFromHome ON PAGE
FOR EACH ROW
WHEN EXISTS ((SELECT MinDistanceFromHome +1
	 FROM PAGE
	 WHERE PageID = new.PageID) < (SELECT MinDistanceFromHome
								       FROM PAGE AS P JOIN HYPERLINK AS H ON 
								       WHERE new.PageID = H.FromPageID and
										     H.ToPageID = P.PageID)
		)
UPDATE ON PAGE
set MinDistanceFromHome = new.MinDistanceFromHome + 1

WHERE PageID = SELECT P.PageID
				FROM  PAGE AS P JOIN HYPERLINK AS H
				JOIN WHERE H.FromPageID = new.PageID and H.ToPageId = P.PageID

		  FROM STATSBYEXAM

BEGIN

			   FROM RECORD
			   WHERE SID = new.SID
